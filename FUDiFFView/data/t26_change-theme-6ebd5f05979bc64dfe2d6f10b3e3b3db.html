var LOAD_THEME_URL="/preference/loadTheme",UPDATE_THEME_URL="/preference/updateTheme";$(document).ready(function(){$("#themeColor").on("change",function(b){b=document.getElementById("themeColor").value;saveTheme(UPDATE_THEME_URL,b);setSessionTheme(b)})}); function setSessionTheme(b){var a=$("body");switch(b){case ui_constants.solid_blue:a.removeClass("pv-bi pv-og");a.addClass("pv-b");break;case ui_constants.gradient_blue:a.removeClass("pv-b pv-og");a.addClass("pv-bi");break;case ui_constants.solid_golden_grey:a.removeClass("pv-bi pv-b");a.addClass("pv-og");break;default:a.removeClass("pv-b pv-og"),a.addClass("pv-bi")}$("input#pvTheme"+b).prop("checked","checked")} function loadTheme(){var b=$("body");$.ajax({url:LOAD_THEME_URL,beforeSend:function(){return sessionStorage.getItem("theme")?(setSessionTheme(sessionStorage.getItem("theme")),!1):!0},success:function(a){200==a.status?(setSessionTheme(a.theme),sessionStorage.setItem("theme",a.theme)):b.addClass("pv-bi")},error:function(){b.addClass("pv-bi")},async:!1})} function saveTheme(b,a){$.ajax({url:b,data:{theme:a},success:function(b){200==b.status?(sessionStorage.setItem("theme",a),console.log("Successfully updated theme")):console.log("Sorry! Couldn't update theme")}})}loadTheme();
